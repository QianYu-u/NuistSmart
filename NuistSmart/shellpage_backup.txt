        private void Scraper_WebMessageReceived(Microsoft.Web.WebView2.Core.CoreWebView2 sender, Microsoft.Web.WebView2.Core.CoreWebView2WebMessageReceivedEventArgs args)
        {
            try
            {
                string message = args.TryGetWebMessageAsString();
                if (message != null && message.StartsWith("YELLOWPAGE_LOG:"))
                {
                    System.Diagnostics.Debug.WriteLine($"[ShellPage] {message}");
                }
                else if (message != null && message.StartsWith("YELLOWPAGE:"))
                {
                    string json = message.Substring("YELLOWPAGE:".Length);
                    var rawList = JsonSerializer.Deserialize<System.Collections.Generic.List<YellowPageRawItem>>(json);
                    
                    if (rawList != null && rawList.Count > 0)
                    {
                        var contacts = new System.Collections.Generic.List<ContactCache>();
                        foreach (var item in rawList)
                        {
                            contacts.Add(new ContactCache
                            {
                                Id = System.Guid.NewGuid().ToString(),
                                Name = item.MC ?? "",
                                Phone = item.PHONE ?? "",
                                WorkPlace = item.WORK_PLACE ?? ""
                            });
                        }
                        App.ServiceProvider.GetRequiredService<NuistSmart.Services.DbService>().SaveContacts(contacts);
                        System.Diagnostics.Debug.WriteLine($"[ShellPage] иғҢжҷҜжҠ“еҸ–йҖҡи®ҜеҪ•жҲҗеҠҹе…ұ {contacts.Count} жқЎи®°еҪ•пјҒ");
                    }
                }
                else if (message != null && message.StartsWith("YELLOWPAGE_ERROR:"))
                {
                    System.Diagnostics.Debug.WriteLine($"[ShellPage] жҠ“еҸ–з»„з»Үж ‘жҠҘй”ҷ: " + message);
                }
            }
            catch (Exception ex)
            {
                System.Diagnostics.Debug.WriteLine($"[ShellPage] и§Јжһҗеӣһдј ж•°жҚ®ејӮеёё: {ex.Message}");
            }
        }

        /// <summary>
        /// жҳҫзӨәеҚ дҪҚеҶ…е®№пјҲеҠҹиғҪжңӘе®һзҺ°ж—¶зҡ„жҸҗзӨәпјү
        /// </summary>
        private void ShowPlaceholder(string message)
        {
            // жё…з©ә Frame еҶ…е®№
            ContentFrame.Content = null;

            // еҲӣе»әдёҖдёӘз®ҖеҚ•зҡ„еҚ дҪҚз•Ңйқў
            var placeholder = new Grid
            {
                HorizontalAlignment = HorizontalAlignment.Center,
                VerticalAlignment = VerticalAlignment.Center
            };

            var stackPanel = new StackPanel
            {
                Spacing = 16,
                HorizontalAlignment = HorizontalAlignment.Center
            };

            var icon = new FontIcon
            {
                Glyph = "\uE946", // Info еӣҫж Ү
                FontSize = 64,
                Foreground = (Microsoft.UI.Xaml.Media.Brush)Application.Current.Resources["AccentFillColorDefaultBrush"]
            };

            var textBlock = new TextBlock
            {
                Text = message,
                FontSize = 18,
                HorizontalAlignment = HorizontalAlignment.Center
            };

            stackPanel.Children.Add(icon);
            stackPanel.Children.Add(textBlock);
            placeholder.Children.Add(stackPanel);

            // е°ҶеҚ дҪҚз•Ңйқўи®ҫзҪ®дёә Frame зҡ„еҶ…е®№
            ContentFrame.Content = placeholder;
        }
